public class FileUploadEmailController{
    public String recipientEmail {get; set;}
    public String subject {get; set;}
    public String body {get; set;}
    
    public Blob uploadedFile {get;set;}
    public String fileName {get;set;}
    
    public boolean isValidEmail(String recipientEmail){
        String regexEmail = '^[A-Za-z0-9._%+-]+@[A-Za-z0-9_-]+\\.[A-Za-z]{2,}$';
  		Pattern pattern =Pattern.compile(regexEmail);
        Matcher matcher = pattern.matcher(recipientEmail);
        return matcher.matches();
       	}
    
    public void sendEmail(){
        if(!isValidEmail(recipientEmail)){
            ApexPages.addMessage(new ApexPages.Message(ApexPages.Severity.ERROR, 'Invalid Email'));
 			return;
        }
            try{
                Messaging.SingleEmailMessage mail = new Messaging.SingleEmailMessage();
                mail.setToAddresses(new String[]{recipientEmail});
                mail.setSubject(subject);
                mail.setPlainTextBody(body);
                if(uploadedFile!=null){
                    Messaging.EmailFileAttachment attachment = new Messaging.EmailFileAttachment();
                    attachment.setFileName(fileName);
                    attachment.setBody(uploadedFile);
                    mail.setFileAttachments( new Messaging.EmailFileAttachment[] {attachment});
                }
                
                Messaging.sendEmail(new Messaging.SingleEmailMessage [] {mail});
            ApexPages.addMessage(new ApexPages.Message(ApexPages.Severity.CONFIRM, 'Email Sent Successfully'));    
            }catch(Exception e){
            ApexPages.addMessage(new ApexPages.Message(ApexPages.Severity.ERROR, 'Error : '+e.getMessage()));
            }
    }
}







<apex:page controller="FileUploadEmailController">
    <apex:form enctype="multipart/form-data">
    <apex:pageMessages />
        <apex:pageBlock title="Email System">       
        <apex:pageBlockSection columns="1">
        <apex:inputText label="Recipient Email" value="{!recipientEmail}" />
        <apex:inputText label="Subject" value="{!subject}" />
        <apex:inputTextarea label="Message Body" value="{!body}" />
		<apex:inputFile value="{!uploadedFile}" fileName="{!fileName}" />
        </apex:pageBlockSection>
            <apex:commandButton value="Send Email" action="{!sendEmail}"  />
		</apex:pageBlock>
    </apex:form>       
</apex:page>
