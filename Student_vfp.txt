public class StController {
    public St__c student {get; set;}
    public List<St__c> studentList {get;set;}
    public Boolean isEdit{get;set;}
    public Decimal selectedRollNo{get;set;}
    
    public StController(){
        student = new St__c();
        studentList = getStudents();
        isEdit=false;
    }
    
    public List<St__c> getStudents(){
        return [SELECT Id , Name,Roll_no__c,Class__c,Number__c from St__c];
    }
    
    public void saveStudent(){
    
    // Mobile validation
    if(student.Number__c != null && String.valueOf((student.Number__c)).length() != 10){
        ApexPages.addMessage(new ApexPages.Message(ApexPages.Severity.ERROR,
            'Mobile number must be 10 digits'));
        return;
    }
    
    // Roll number validatio
    if(student.Roll_no__c != null && student.Roll_no__c <= 0){
        ApexPages.addMessage(new ApexPages.Message(ApexPages.Severity.ERROR,
            'Roll number must be greater than 0'));
        return;
    }

    try{
        if(isEdit){
            update student;
        } else {
            insert student;
        }

        ApexPages.addMessage(new ApexPages.Message(
            ApexPages.Severity.CONFIRM,
            isEdit ? 'Record updated successfully' : 'Record saved successfully'
        ));

        student = new St__c();
        isEdit = false;
        studentList = getStudents();

    } catch(DmlException e){
        ApexPages.addMessage(new ApexPages.Message(
            ApexPages.Severity.ERROR,
            'Duplicate Roll Number not allowed'
        ));
    }
}
    
    public void editStudent(){
        student = [SELECT Id , Name,Roll_no__c,Class__c ,Number__c from St__c where Roll_no__c=:selectedRollNo LIMIT 1];
        isEdit=true;
    }
    
    public void deleteStudent(){
        St__c s=[SELECT Id from St__c where Roll_no__c =:selectedRollNo LIMIT 1];
        delete s;
        studentList = getStudents();
    }
}
